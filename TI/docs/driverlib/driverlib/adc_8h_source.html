565 __STATIC_INLINE uint32_t ADCReadResult(uint32_t index)
566 {
567 
568  while (HWREG(ADC_BASE + ADC_O_STA) & ADC_STA_BUSY_ACTIVE) {}
569 
570  /* Return data from result register */
571  return HWREG(ADC_BASE + ADC_O_MEMRES0 + (4 * index));
572 }
